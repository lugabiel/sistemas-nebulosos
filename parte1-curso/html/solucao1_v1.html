Operador Soma Drástica

sDrastica = zeros(1001)';
x=(0:0.01:10)';
a= trimf(x,[1,4,7]);
b=gaussmf(x,[1,2]);

for p = 1: length(x)
    if min(a(p),b(p)) == 0
        sDrastica(p) = max(a(p),b(p));
    else
        sDrastica(p) = 1;
    end
end
plot(x,a, ...
    x,b, ...
    x,sDrastica,'.-');
hXLabel = xlabel('valor');

    'Conjunto A', ...

    'A Soma Drástica B');
